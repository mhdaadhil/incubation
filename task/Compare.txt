public class Compare{
public static void main{
 Scanner sc =new Scanner (System.in);

int a=sc.nextInt();
int b=sc.nextInt();

if (a<b)
System.out.println(a+"<"+b);

if (a>b)
 System.out.println(a+">"+b);

if (a>=b)
System.out.println(a+">="+b);

if (a<=b)
System.out.println(a+"<="+b);

if (a==b)
System.out.println(a+"=="+b);

if (a!=b)
System.out.println(a+"!="+b);

}
}
